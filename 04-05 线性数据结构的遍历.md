#  线性数据结构的遍历

> 遍历：将集合中的每一个元素进行获取并查看

> 循环遍历如下：

```js
var arr = [1,2,3,4,5,6,7,8];

function bianArr(arr) {
    if (arr == null) return; //算法必须有判空处理，因为如果没有判空，arr.length就会报错，算法报错就没有分
    for (var i = 0 ; i < arr.length ; i ++) {
        console.log(arr[i]);
    }
}

bianArr(arr); //1 2 3 4 5 6 7 8

function Node(value) {
    this.value = value;
    this.next = null;
}

var node1 = new Node(1);
var node2 = new Node(2);
var node3 = new Node(3);
var node4 = new Node(4);
var node5 = new Node(5);

node1.next = node2;
node2.next = node3;
node3.next = node4;
node4.next = node5;

function bianLink(root) {//root在数据结构中代表根节点
    var temp = root;
    while(true) {
        if (temp != null) {
            console.log(temp.value);
        } else {
            break;
        }
        temp = temp.next;
    }
}

bianLink(node1); //1 2 3 4 5
```

> 递归遍历如下：（循环遍历是更好的处理方式，但是针对链表有些时候我们常常使用递归遍历）

```js
var arr = [1,2,3,4,5,6,7,8];

function bianArr(arr, i) {
    if (arr == null || arr.length <= i) return;
    console.log(arr[i]);
    bianArr(arr, i + 1);
}
bianArr(arr, 0);


function Node (value) {
    this.value = value;
    this.next = null;
}

var node1 = new Node(1);
var node2 = new Node(2);
var node3 = new Node(3);
var node4 = new Node(4);
var node5 = new Node(5);

node1.next = node2;
node2.next = node3;
node3.next = node4;
node4.next = node5;

//递归遍历，必须有出口
function bianLink(root) {
    if (root == null) return;
    console.log(root.value);
    bianLink(root.next);
}

// bianLink(node1);
```
