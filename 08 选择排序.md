# 选择排序

> 选择排序的核心：每次将最大的一个数选出来，并和最后一个位置的数进行交换

```js
//选择排序
const arr = [3, 7, 9, 6, 5, 2, 4, 1, 8, 5];

//比较
function compare(a, b) {
  if (a > b) {
    return true;
  }
}

//交换
function exchange(arr, index1, index2) {
  const temp = arr[index1];
  arr[index1] = arr[index2];
  arr[index2] = temp;
}

function sort(arr) {
  for (let i = 0; i < arr.length; i++) {
    let maxIndex = 0;
    for (let j = 0; j < arr.length - i; j++) {
      if (compare(arr[j], arr[maxIndex])) {
        maxIndex = j;
      }
    }
    exchange(arr, maxIndex, arr.length - 1 - i);
  }
}

sort(arr);

console.log(arr);
```
