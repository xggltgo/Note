# 二叉树的前序遍历

二叉树结构如下：

```tex
		A
	B		 C
D  E	F  G
```

中序遍历：DBEAFCG

代码实现：

```js
function Node(val) {
  this.val = val;
  this.left = null;
  this.right = null;
}

const A = new Node("A");
const B = new Node("B");
const C = new Node("C");
const D = new Node("D");
const E = new Node("E");
const F = new Node("F");
const G = new Node("G");

A.left = B;
A.right = C;
B.left = D;
B.right = E;
C.left = F;
C.right = G;

function ergodic(root) {
  if (root === null) {
    return;
  }
  ergodic(root.left);
  console.log(root.val);
  ergodic(root.right);
}

ergodic(A);

// 输出如下：
// D
// B
// E
// A
// F
// C
// G
```
