# 代码实现前序中序还原二叉树

前序遍历（先根次序遍历）：根左右 ABDECFG

中序遍历（中根次序遍历）：左根右 DBEAFCG

1. 由前序遍历找到根节点 A
2. 根据中序遍历中的根节点位置,确定左子树和右子树

```js
const preorder = ["A", "B", "D", "E", "C", "F", "G"];
const inOrder = ["D", "B", "E", "A", "F", "C", "G"];

function Node(val) {
  this.val = val;
  this.left = null;
  this.right = null;
}

function revert(preorder, inOrder) {
  if (
    preorder === null ||
    inOrder === null ||
    preorder.length === 0 ||
    inOrder.length === 0
  ) {
    return;
  }
  // 1.找到二叉树的根节点
  const root = preorder[0];
  const ROOT = new Node(root);
  // 2.分析出左子树
  const index = inOrder.indexOf(root);
  const preorder_leftChildTree = preorder.slice(1, index + 1);
  const preorder_rightChildTree = preorder.slice(index + 1);
  const inOrder_leftChildTree = inOrder.slice(0, index);
  const inOrder_rightChildTree = inOrder.slice(index + 1);
  ROOT.left = revert(preorder_leftChildTree, inOrder_leftChildTree);
  ROOT.right = revert(preorder_rightChildTree, inOrder_rightChildTree);
  return ROOT;
}

const root = revert(preorder, inOrder);
console.log(root);
```
