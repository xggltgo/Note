## Carousel 轮播图

```js
import React, { useRef } from 'react';
import { Carousel, Button } from 'antd';
import img1 from '../../assets/img/1.png';
import img2 from '../../assets/img/2.png';
import img3 from '../../assets/img/3.png';
import styles from './index.css';

const ButtonGroup = Button.Group;

export default function MyCarousel() {
  const carouselRef = useRef();
  const carouselItem = [img1, img2, img3].map((item, index) => (
    <div className={styles['img-container']} key={index}>
      <img src={item} alt="" />
    </div>
  ));

  return (
    <div className={styles.container}>
      <ButtonGroup>
        <Button
          onClick={() => {
            carouselRef.current.prev();
          }}
        >
          prev
        </Button>
        <Button
          onClick={() => {
            carouselRef.current.next();
          }}
        >
          next
        </Button>
      </ButtonGroup>
      <Carousel ref={carouselRef}>{carouselItem}</Carousel>
    </div>
  );
}

```

## Modal 对话框

```js
import React, { useState } from 'react';
import { Modal, Button } from 'antd';

export default function MyModal() {
  const [visible, setVisible] = useState(false);

  function handleOk(e) {
    console.log(e);
    setVisible(false);
  }
  function handleCancel(e) {
    console.log(e);
    setVisible(false);
  }
  return (
    <>
      <Button
        type="primary"
        onClick={() => {
          setVisible(true);
        }}
      >
        modal
      </Button>
      <Modal title="Test Modal" visible={visible} onOk={handleOk} onCancel={handleCancel}>
        <p>123...</p>
        <p>abc...</p>
        <p>+-/...</p>
      </Modal>
    </>
  );
}
```
