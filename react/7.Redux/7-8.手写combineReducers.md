# 手写combineReducers

**核心思想**

组装reducers（*个plain-object 该对象每个属性对应一个reducer函数*），

返回一个reducer函数，该函数的状态为一个对象 

该对象记录了每一个reducer对应的数据，对象的属性名与传递的参数对象保持一致

```js
// combineResucers

import isPlainObject from './utils/isPlainObject';
import specialActionTypes from './utils/specialActionTypes';

/**
 * 返回一个reducer函数 该函数的状态为一个对象 该对象记录了每一个reducer对应的数据
 * @param {Object} reducers 一个plain-object 该对象每个属性对应一个reducer函数
 */
export default function (reducers) {
  //1. 验证
  validateReducers(reducers);
  /**
     * 返回的是一个reducer函数
     */
  return function (state = {}, action) {
    let newState = {}; //要返回的新的状态
    for (const key in reducers) {
      if (Object.hasOwnProperty.call(reducers, key)) {
        const reducer = reducers[key];
        newState[key] = reducer(state[key], action);
      }
    }
    return newState; //返回状态
  };
}

function validateReducers(reducers) {
  if (typeof reducers !== 'object') {
    throw new TypeError('reducers must be an object');
  }
  if (!isPlainObject(reducers)) {
    throw new TypeError('reducers must be a plain objects');
  }
  //验证reducer的返回结果是不是undefined
  for (const key in reducers) {
    if (Object.hasOwnProperty.call(reducers, key)) {
      const reducer = reducers[key]; //拿到reducer
      //传递一个特殊的type值
      let state = reducer(undefined, { type: specialActionTypes.INIT() });
      if (state === undefined) {
        throw new Error('reducer can not return undefined');
      }
      state = reducer(undefined, { type: specialActionTypes.UNKNOWN() });
      if (state === undefined) {
        throw new Error('reducer can not return undefined');
      }
    }
  }
}
```
