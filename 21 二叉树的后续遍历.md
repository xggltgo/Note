# 二叉树的前序遍历

二叉树结构如下：

```tex
		A
	B		 C
D  E	F  G
```

后序遍历：DEBFGCA

代码实现：

```js
function Node(val) {
  this.val = val;
  this.left = null;
  this.right = null;
}

const A = new Node("A");
const B = new Node("B");
const C = new Node("C");
const D = new Node("D");
const E = new Node("E");
const F = new Node("F");
const G = new Node("G");

A.left = B;
A.right = C;
B.left = D;
B.right = E;
C.left = F;
C.right = G;

function ergodic(root) {
  if (root === null) {
    return;
  }
  ergodic(root.left);
  ergodic(root.right);
  console.log(root.val);
}

ergodic(A);

// 输出如下：
// D
// E
// B
// F
// G
// C
// A
```
