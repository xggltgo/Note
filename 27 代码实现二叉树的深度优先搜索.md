# 代码实现二叉树的深度优先搜索

> 对于二叉树，其深度优先搜索和前序遍历的顺序是相同的

```js
function Node(val) {
  this.val = val;
  this.left = null;
  this.right = null;
}

const A = new Node('A');
const B = new Node('B');
const C = new Node('C');
const D = new Node('D');
const E = new Node('E');
const F = new Node('F');
const G = new Node('G');

A.left = B;
A.right = C;
B.left = D;
B.right = E;
C.left = F;
C.right = G;

function depthFirstSearch(binaryTree, nodeVal) {
  if (binaryTree === null) {
    return false;
  }
  if (binaryTree.val === nodeVal) {
    return true;
  }
  return (
    depthFirstSearch(binaryTree.left, nodeVal) ||
    depthFirstSearch(binaryTree.right, nodeVal)
  );
}

console.log(depthFirstSearch(A, 'A'));
console.log(depthFirstSearch(A, 'B'));
console.log(depthFirstSearch(A, 'C'));
console.log(depthFirstSearch(A, 'D'));
console.log(depthFirstSearch(A, 'E'));
console.log(depthFirstSearch(A, 'F'));
console.log(depthFirstSearch(A, 'G'));
console.log(depthFirstSearch(A, 'H'));

// 输出结果如下：
