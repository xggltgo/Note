# Cascader级联选择

```js
import React, { useEffect, useState } from 'react';
import { Cascader } from 'antd';
import { getProvinceInfo } from '../../api/province';
function CascaderComp(props, ref) {
  const [options, setOptions] = useState([]);
  useEffect(() => {
    getProvinceInfo().then(res => {
      setOptions(res);
    });
  }, []);
  return (
    <Cascader
      onChange={value => {
        props.onChange && props.onChange(value);
      }}
      value={props.value}
      ref={ref}
      options={options}
      placeholder="Please select"
    />
  );
}

export default React.forwardRef(CascaderComp);
```
