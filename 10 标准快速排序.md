# 标准快速排序

```js
const origin = [3, 5, 7, 1, 2, 4, 6, 8];

// 快排思想：首尾指针向中靠拢，将首指针指向的值与尾指针指向的值进行比较，首指针指向的值如果大于尾指针指向的值，则交换其位置

function sort(arr) {
  _sort(arr, 0, arr.length - 1);
  // 辅助函数
  function _sort(arr, start, end) {
    if (start >= end) {
      return;
    }
    let _start = start + 1;
    let _end = end;
    const standard = arr[start]; //1
    while (_start < _end) {
      // 首指针找到比第一个元素大的值
      while (arr[_start] < standard && _start < _end) {
        _start += 1;
      }
      // 尾指针找到比第一个元素小的值
      while (arr[_end] > standard && _start < _end) {
        _end -= 1;
      }
      // 交换位置
      if (_start < _end) {
        const temp = arr[_start];
        arr[_start] = arr[_end];
        arr[_end] = temp;
      }
    }
    const index = _start - 1;
    const temp = arr[start];
    arr[start] = arr[index];
    arr[index] = temp;
    _sort(arr, start, index);
    _sort(arr, index + 1, arr.length - 1);
  }
}

sort(origin);
console.log(origin);
```
