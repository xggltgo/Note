# 二叉树的比较

> 判断两颗二叉树是否相同

二叉树节题核心：对比根结点后，递归对比左子树和右子树

```js
function Node(val) {
  this.val = val;
  this.left = null;
  this.right = null;
}

const A1 = new Node('A');
const B1 = new Node('B');
const C1 = new Node('C');
const D1 = new Node('D');
const E1 = new Node('E');
const F1 = new Node('F');
const G1 = new Node('G');

A1.left = B1;
A1.right = C1;
B1.left = D1;
// B1.right = E1;
C1.left = F1;
C1.right = G1;

const A2 = new Node('A');
const B2 = new Node('B');
const C2 = new Node('C');
const D2 = new Node('D');
const E2 = new Node('E');
const F2 = new Node('F');
const G2 = new Node('G');

A2.left = B2;
A2.right = C2;
B2.left = D2;
// B2.right = E2;
C2.left = F2;
C2.right = G2;

function binaryTreeCompare(tree1, tree2) {
  if (tree1 === tree2) {
    return true;
  }
  if (
    (tree1 === null && tree2) ||
    (tree1 && tree2 === null) ||
    tree1.val !== tree2.val
  ) {
    return false;
  }
  return (
    binaryTreeCompare(tree1.left, tree2.left) &&
    binaryTreeCompare(tree1.right, tree2.right)
  );
}

console.log(binaryTreeCompare(A1,A2));
```
